&&(|v_1v_2|+\dots+|v_{n-1}v_n|+|v_nv_1|)^2 \\
&\le& (1^2+1^2+\dots+1^2)\cdot (|v_1v_2|^2+\dots+|v_{n-1}v_n|^2+|v_nv_1|^2) \\
&=& n\cdot (|v_1v_2|^2+\dots+|v_{n-1}v_n|^2+|v_nv_1|^2) \\
&\le& n\cdot 4
\end{eqnarray*}
\end{enumerate}
